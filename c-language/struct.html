構造体

struct 構造体タグ名{
    データ型1 メンバ名1;
    データ型2 メンバ名2;
                 :
    データ型n メンバ名n;
};

#include <stdio.h>
struct Person{
    char name[20];
    int age;
};

int main(void){
    struct Person p = {"Alice", 20};
    struct Person *ptr = &p;

    printf("%s is %d years old.\n", ptr->name, ptr->age);
    return 0;
}
