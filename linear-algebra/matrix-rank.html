\[ \begin{align} A&=\begin{bmatrix} 2 & 0 & 1 \\ 1 & -3 & 2 \\ -1 & 1 & -1 \end{bmatrix}\to \begin{bmatrix} 1 & -3 & 2 \\ 2 & 0 & 1 \\ -1 & 1 & -1 \end{bmatrix}\to \begin{bmatrix} 1 & -3 & 2 \\ 0 & 6 & -3 \\ 0 & -2 & 1 \end{bmatrix}\\ &\to\begin{bmatrix} 1 & -3 & 2 \\ 0 & 1 & -\frac{1}{2} \\ 0 & -2 & 1 \end{bmatrix}\to \begin{bmatrix} 1 & 0 & \frac{3}{2} \\ 0 & 1 & -\frac{1}{2} \\ 0 & 0 & 0 \end{bmatrix} \end{align} \]
